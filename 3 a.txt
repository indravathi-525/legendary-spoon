
#include <stdio.h>
#include <stdlib.h>
int main()
{
    char c[1000];
    FILE *fptr;
fptr=fopen("a.txt","r");
    if ((fptr == NULL)
    {
        printf("Error! opening file");
        exit(1);         
    }

    // reads text until newline 
    fscanf(fptr,"%[^\n]", c);
if(strlen(c)>100)
{
int i=0;
do
    {
        char c = fgetc(fptr); 
        if (i==100) 
            break ; 
        printf("%c", c); 
i++;
    }  while(1); 
fclose(fptr);
}
else
    printf("Data from the file:\n%s", c);
    fclose(fptr);
    
    return 0;
}
